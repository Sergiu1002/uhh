#include <iostream>
#include<fstream>
using namespace std;

int n, v[10], e, sumac = 0, temp, suman = 0, contor = 0, sp = 0, x[10], s;

ifstream f("intrare.txt");
ofstream g("iesire.txt");

void afisare(int n)
{
	for (int i = 1; i <= n; i++)
		if (x[i] != 0)
			g << v[x[i]] << " ";
	g << endl;
}

int valid(int k)
{
	if (k > 1)
		if (v[x[k]] > v[x[k - 1]]) return 0;
	return 1;
}

void backtracking(int x[], int k, int s)
{
	int i;
	for (i = x[k - 1] + 1; i <= n; i++)
	{
		x[k] = i;
		s = s + v[x[k]];
		if (valid(k))
			if (s >= 45) afisare(k);
			else backtracking(x, k + 1, s);
		s = s - v[x[k]];
	}
}

int main()
{
	f >> n;
	for (int i = 1; i <= n; i++)
		f >> v[i];
	f >> e;
	//nota 3

	for (int i = 1; i <= n; i++)
		sumac += v[i];

	if (sumac = 60)
	{
		n += 1;
		v[n] = 60 - sumac;
	}
	//nota 4

	for (int i = 1; i <= n; i++)
	{
		for (int j = 1 + i; j <= n; j++)
		{
			if (v[i] < v[j])
			{
				temp = v[i];
				v[i] = v[j];
				v[j] = temp;
			}
		}
	}
	for (int i = 1; i <= n; i++)
	{
		if (suman < 45)
		{
			suman += v[i];
			cout << "Disciplina " << i << " " << v[i] << endl;
			contor++;
		}
	}
	cout << "Studentul trebuie sa promoveze " << contor << " examene pentru a promova care echivaleaza " << suman << " credite (>45)";
	//nota 6
	backtracking(x, 1, 0);
}
